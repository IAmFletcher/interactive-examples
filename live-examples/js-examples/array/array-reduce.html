const array1 = [1, 2, 3, 4];

// 1 + 2 + 3 + 4
const result1 = array1.reduce((accumulator, currentValue) => accumulator + currentValue);
console.log(result1);
// expected output: 10

// 5 + 1 + 2 + 3 + 4
const result2 = array1.reduce((accumulator, currentValue) => accumulator + currentValue, 5);
console.log(result2);
// expected output: 15
